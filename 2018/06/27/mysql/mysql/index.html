MySQL安装指南

发表于 2018-06-27 | 分类于 MySQL

ubuntu install mysql

apt-get install mysql-server

root设置密码

service mysql stop

sudo mkdir -p /var/run/mysqld
sudo chown mysql:mysql /var/run/mysqld

sudo mysqld_safe --skip-grant-tables &

mysql -u root

use mysql

UPDATE user SET authentication_string = PASSWORD('NewPassword') , password_expired = 'N' WHERE User = 'root' AND Host = 'localhost';

update user set plugin="mysql_native_password";

flush privileges;

quit

sudo kill -9 $(pgrep mysql)

service mysql start
